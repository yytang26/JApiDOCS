public class ApiResult implements Serializable{

    private UserVO[] body; //
    private int code; //
    private String errMsg; //
    private UserVO[] data; //
    private enum errType; // [SUCCESS,WARN,ERROR]

    public UserVO[] getBody(){
        return body;
    }

    public void setBody(UserVO[] body){
        this.body = body;
    }

    public int getCode(){
        return code;
    }

    public void setCode(int code){
        this.code = code;
    }

    public String getErrMsg(){
        return errMsg;
    }

    public void setErrMsg(String errMsg){
        this.errMsg = errMsg;
    }

    public UserVO[] getData(){
        return data;
    }

    public void setData(UserVO[] data){
        this.data = data;
    }

    public enum getErrType(){
        return errType;
    }

    public void setErrType(enum errType){
        this.errType = errType;
    }

}
